// ────────────────────────────────────────────────

//  Real-time FSD impact calculation

// ────────────────────────────────────────────────

const START_MILES = 8.62e9;               // March 11, 2026 estimate

const MILES_PER_DAY = 20e6;

const FSD_CRASH_RATE = 5.3e6;             // miles per crash

const HUMAN_CRASH_RATE = 660e3;

const FATALITY_RATE_PER_CRASH = 0.006;

const AVG_ACCIDENT_COST = 56000;

const VSL = 13700000;                     // Value of Statistical Life


let start = Date.now() / 1000;


function updateStats() {

    const elapsedSec = (Date.now() / 1000) - start;

    const miles = START_MILES + (elapsedSec / 86400) * MILES_PER_DAY;


    const expectedCrashes = miles / HUMAN_CRASH_RATE;

    const actualCrashes   = miles / FSD_CRASH_RATE;

    const avoided         = expectedCrashes - actualCrashes;

    const lives           = avoided * FATALITY_RATE_PER_CRASH;

    const dollars         = avoided * AVG_ACCIDENT_COST + lives * VSL;


    document.getElementById("lives").textContent     = Math.round(lives).toLocaleString();

    document.getElementById("accidents").textContent = Math.round(avoided).toLocaleString();

    document.getElementById("savings").textContent   = (dollars / 1e9).toFixed(1) + " B";

}


setInterval(updateStats, 1300);

updateStats();
